% Tumor Cell State Machine
\begin{tikzpicture}[
  state/.style={draw, rounded corners=5pt, minimum width=2.4cm, minimum height=1.1cm,
                align=center, font=\small, thick},
  trans/.style={font=\footnotesize, align=center, inner sep=1pt},
  arrow/.style={->, >=stealth, thick},
  deatharrow/.style={->, >=stealth, thick, red!60},
]

% ---- States ----
\node[state, fill=blue!10] (Q) at (0, 4) {\textbf{Quiescent}\\dormant};
\node[state, fill=green!15] (A) at (4.5, 4) {\textbf{Active}\\high metabolism};
\node[state, fill=orange!15] (D) at (9, 4) {\textbf{Dividing}\\mitosis};
\node[state, fill=purple!15] (G) at (4.5, 0.5) {\textbf{Aggressive}\\invasion};
\node[state, fill=red!25] (X) at (0, -2.5) {\textbf{Apoptotic}\\cell death};

% ---- Main transitions ----
% Quiescent <-> Active
\draw[arrow] (Q.east) -- (A.west)
  node[trans, above, midway] {nutrients\\available};
\draw[arrow] (A.west) ++(-0.05, -0.2) -- ++(-3.85, 0)
  node[trans, below, midway] {nutrient\\depletion};

% Active -> Dividing -> Active
\draw[arrow] (A.east) -- (D.west)
  node[trans, above, midway] {DNA ready};
\draw[arrow] (D.south west) to[bend right=20] (A.north east)
  node[trans, below, sloped, midway] {division done};

% Active <-> Aggressive
\draw[arrow] (A.south) -- (G.north)
  node[trans, right, midway] {hypoxia / stress};
\draw[arrow] (G.north) ++(0.05, 0) -- ++(0, 2.35)
  node[trans, left=5pt, midway] {O$_2$ restored};

% ---- Death transitions (all -> Apoptotic) ----
\draw[deatharrow] (Q.south) -- (X.north)
  node[trans, left, midway] {DNA\\damage};
\draw[deatharrow] (A.south west) -- (X.north east)
  node[trans, above, sloped, midway] {immune kill};
\draw[deatharrow] (D.south) to[bend left=15] (X.east)
  node[trans, right, midway] {mitotic\\arrest};
\draw[deatharrow] (G.west) -- (X.south east)
  node[trans, below, sloped, midway] {treatment};

% ---- Key Parameters box ----
\node[draw, rounded corners=3pt, fill=blue!5, inner sep=5pt,
      text width=3.6cm, align=left, font=\footnotesize] (params) at (9, 0.5) {%
\textbf{Key Parameters}\\[2pt]
\texttt{tumour\_growth\_effect}\\
\texttt{tumour\_apoptosis\_effect}\\
DNA mutations\\
Glucose availability};

\end{tikzpicture}
