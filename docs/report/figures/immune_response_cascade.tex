% Immune Response Cascade Diagram for Chapter 2
\begin{tikzpicture}[scale=0.9, every node/.style={scale=0.9}]

% Define colors for immune cell types
\definecolor{innate}{RGB}{255,165,0}
\definecolor{adaptive}{RGB}{30,144,255}
\definecolor{tumor}{RGB}{220,20,60}
\definecolor{suppressor}{RGB}{128,128,128}

% Timeline
\draw[thick, ->] (0,0) -- (14,0);
\node[below] at (14,0) {Time};

% Phases
\node[above] at (2,0) {\textbf{Recognition}};
\node[above] at (6,0) {\textbf{Activation}};
\node[above] at (10,0) {\textbf{Effector}};
\node[above] at (13,0) {\textbf{Resolution}};

% Phase separators
\foreach \x in {4, 8, 11.5} {
    \draw[dashed, gray] (\x,-3) -- (\x,8);
}

% Innate immune response (early)
\node[draw, circle, fill=innate, text=white, minimum size=1cm] (nk) at (1,6) {NK};
\node[draw, circle, fill=innate, text=white, minimum size=0.8cm] (macro) at (2,4) {M$\phi$};
\node[draw, circle, fill=innate, text=white, minimum size=0.8cm] (dc) at (3,5) {DC};

% Antigen presentation
\draw[thick, ->] (dc) -- (5,5) node[midway, above] {Ag presentation};

% Adaptive immune response (later)
\node[draw, circle, fill=adaptive, text=white, minimum size=0.8cm] (cd8naive) at (5,5) {CD8$^+$};
\node[draw, circle, fill=adaptive, text=white, minimum size=0.8cm] (cd4naive) at (5,3) {CD4$^+$};

% T cell differentiation
\draw[thick, ->] (cd8naive) -- (8,6) node[midway, above] {Activation};
\draw[thick, ->] (cd4naive) -- (8,4) node[midway, above] {Th1};
\draw[thick, ->] (cd4naive) -- (8,2) node[midway, below] {Th2};

% Effector cells
\node[draw, circle, fill=adaptive, text=white, minimum size=1cm] (ctl) at (8,6) {CTL};
\node[draw, circle, fill=adaptive, text=white, minimum size=0.8cm] (th1) at (8,4) {Th1};
\node[draw, circle, fill=adaptive, text=white, minimum size=0.8cm] (th2) at (8,2) {Th2};

% Tumor interaction
\node[draw, star, fill=tumor, text=white, minimum size=1.2cm] (tumor_cell) at (11,4) {Tumor};

% Cytotoxic killing
\draw[thick, ->, red] (ctl) -- (tumor_cell) node[midway, above] {Kill};
\draw[thick, ->, red] (nk) to[bend left=30] (tumor_cell);

% Helper functions
\draw[thick, ->, blue] (th1) -- (tumor_cell) node[midway, below right] {Support};

% Immunosuppressive elements
\node[draw, circle, fill=suppressor, text=white, minimum size=0.8cm] (treg) at (6,1) {Treg};
\node[draw, circle, fill=suppressor, text=white, minimum size=0.8cm] (m2) at (10,1) {M2};

% Suppression
\draw[thick, ->, dashed, suppressor] (treg) to[bend right=30] (ctl) node[midway, below] {Suppress};
\draw[thick, ->, dashed, suppressor] (m2) -- (tumor_cell) node[midway, below] {Protect};

% Memory formation
\node[draw, circle, fill=adaptive!50, text=white, minimum size=0.8cm] (memory) at (12,6) {T$_M$};
\draw[thick, ->] (ctl) -- (memory) node[midway, above] {Memory};

% Cytokine network
\node[draw, diamond, fill=yellow!80, minimum size=0.6cm] (ifng) at (9,5) {IFN-$\gamma$};
\node[draw, diamond, fill=yellow!80, minimum size=0.6cm] (il2) at (9,3.5) {IL-2};
\node[draw, diamond, fill=yellow!80, minimum size=0.6cm] (tnfa) at (9,2.5) {TNF-$\alpha$};

% Cytokine production
\draw[dashed, ->] (ctl) -- (ifng);
\draw[dashed, ->] (th1) -- (il2);
\draw[dashed, ->] (th1) -- (tnfa);

% Title and legend
\node[text width=14cm, text centered] at (7,8.5) {
    \Large\textbf{Immune Response Cascade in RCC}
};

% Legend
\node[draw, rounded corners, fill=white, text width=3cm] at (1,7.5) {
    \textbf{Legend}\\
    \textcolor{innate}{\textbullet} Innate\\
    \textcolor{adaptive}{\textbullet} Adaptive\\
    \textcolor{tumor}{\textbullet} Tumor\\
    \textcolor{suppressor}{\textbullet} Suppressor\\
    \textcolor{yellow!80}{\textbullet} Cytokines
};

\end{tikzpicture}
